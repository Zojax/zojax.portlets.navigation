    >>> actor = Browser()
    >>> actor.handleErrors = False
    >>> actor.addHeader("Authorization", "Basic mgr:mgrpw")
    >>> actor.open("http://localhost/++extensions++/")
    >>> actor.getLink('Portlets').click()

    >>> actor.getLink('Left column portlets').click()
